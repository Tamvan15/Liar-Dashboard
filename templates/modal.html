Contoh Penggunaan API:

curl -X 'POST' \
  'http://127.0.0.1:8000/api-token-auth/' \
  -H 'accept: application/json' \
  -d '{
  "username": "liar",
  "password": "liar"
}'

# Respon:
# {"token":"xxxxxxxxxxxxxxxxxxx"}

# Gunakan token untuk mengakses API:
curl -X 'GET' \
  'http://127.0.0.1:8000/api/tables/' \
  -H 'accept: application/json' \
  -H 'Authorization: Token xxxxxxxxxxxxxxxxxxxxx'

# Untuk mengakses tabel tertentu:
curl -X 'GET' \
  'http://127.0.0.1:8000/api/tables/1/' \
  -H 'accept: application/json' \
  -H 'Authorization: Token xxxxxxxxxxxxxxxxxxxxx'
